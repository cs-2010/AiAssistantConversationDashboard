        // Configure Rigidbody2D for horizontal-only movement
        rb.gravityScale = 0f;
        rb.constraints = RigidbodyConstraints2D.FreezeRotation | RigidbodyConstraints2D.FreezePositionY;
        
        // Set the velocity to move left with doubled speed
        rb.velocity = Vector2.left * doubledSpeed;
